// === Función para calcular PWMA ===
pwma(src, length) =>
    var float sum_weight = 0.0
    var float sum_price_weight = 0.0
    
    sum_weight := 0.0
    sum_price_weight := 0.0

    for i = 0 to length - 1
        weight = math.pow(i + 1, 2) // peso parabólico
        price  = src[length - 1 - i] // desde el más antiguo hasta el más reciente
        sum_weight += weight
        sum_price_weight += price * weight

    sum_price_weight / sum_weight

// === Cálculo ===
pwmaValue = pwma(src, length)
diff      = src - pwmaValue

// === Panel 1: Precio y PWMA ===
pricePlot = plot(src, color=color.blue, title="Precio", transp=90, display=labelDisplay)
pwmaPlot  = plot(pwmaValue, color=colorPwma, linewidth=widthPwma, title="PWMA", display=labelDisplay)

fill(pricePlot, pwmaPlot, color = fillEnabled ? (src > pwmaValue ? color.new(color.green, 85) : color.new(color.red, 85)) : color.new(color.white, 100))

power = input.int(title='Power', minval=0, defval=2)
highlightMovements = input(title='Highlight Movements ?', defval=true)
//src = input(title='Source', defval=close)

sum = 0.0
weightSum = 0.0

for i = 0 to length - 1 by 1
    weight = math.pow(length - i, power)
    sum += nz(src[i]) * weight
    weightSum += weight
    weightSum

pwma = sum / weightSum

pwmaColor = highlightMovements ? pwma > pwma[1] ? color.green : color.red : colorPwma
plot(pwma, title='PWMA', linewidth=widthPwma, color=pwmaColor, transp=0, display=labelDisplay)
