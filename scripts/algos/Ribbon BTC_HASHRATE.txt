strategy('<[Osc]Hash Ribbons Strategy[BTC]>', shorttitle='*Ribbon BTC_HASHRATE*', precision=8, initial_capital=1000)

// NOTES
// Updated source code
// The "Spring" is the confirmed Miner capitulation period:
// - The 1st "gray" circle is the start of Capitulation (1 month Hash Rate crosses UNDER 2 month Hash Rate)
// - Last "green" circle is the end of Capitulation (1 month Hash Rate crosses OVER 2 month Hash Rate)
// - The "greener" the spring gets (up until blue) represents Hash Rate recovery (it is increasing)
// - The "blue" circle is the first instance of positive momentum following recovery of Hash Rate (1m HR > 2m HR). This is historically a rewarding place to buy with limited downside.

// INPUTS
type = input.string('Ribbons', options=['Ribbons', 'Oscillator'], title='Plot Type')
len_s = input(30, 'Hash Rate Short SMA (days).')
len_l = input(60, 'Hash Rate Long SMA (days).')
signals = input(true, 'Plot Signals')
plot_halvings = input(true, 'Plot Halvings')
raw = input(true, 'Plot Raw Hash Rate')

// inputs
trade_leverage = input.float(1, title='Trade - Leverage', step=0.25)
trade_stop = input.float(20, title='Trade - Stop Loss Percent', step=0.1, minval=0.1)
trade_risk = input.float(100, title='Trade - Risk Percent', step=0.1, minval=0.1, maxval=100)

// INPUT BACKTEST RANGE 
FromMonth = input.int(defval=1, title='From Month', minval=1, maxval=12)
FromDay = input.int(defval=1, title='From Day', minval=1, maxval=31)
FromYear = input.int(defval=2015, title='From Year', minval=2012)
ToMonth = input.int(defval=1, title='To Month', minval=1, maxval=12)
ToDay = input.int(defval=1, title='To Day', minval=1, maxval=31)
ToYear = input.int(defval=9999, title='To Year', minval=2017)

// FUNCTION EXAMPLE 
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() =>
    time >= start and time <= finish ? true : false

// Functions
get_round(value, precision) =>
    math.round(value * math.pow(10, precision)) / math.pow(10, precision)

// HASH RATE MA
// HR on TV only has "yesterday's" value --> use "lookahead_on" when running live (on current bar), to pull forward yesterdays data
live_HR_raw = request.security('BTC_HASHRATE', 'D', close, gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_on)
live_HR_short = request.security('BTC_HASHRATE', 'D', ta.sma(close, len_s), gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_on)
live_HR_long = request.security('BTC_HASHRATE', 'D', ta.sma(close, len_l), gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_on)

hist_HR_raw = request.security('BTC_HASHRATE', 'D', close, gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_off)
hist_HR_short = request.security('BTC_HASHRATE', 'D', ta.sma(close, len_s), gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_off)
hist_HR_long = request.security('BTC_HASHRATE', 'D', ta.sma(close, len_l), gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_off)

daily_s10 = request.security(syminfo.tickerid, 'D', ta.sma(close, 10), gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_off)
daily_s20 = request.security(syminfo.tickerid, 'D', ta.sma(close, 20), gaps=barmerge.gaps_off, lookahead=barmerge.lookahead_off)

// DAILY TIMEFRAME MGMT
is_newbar(res) =>
    t = time(res)
    ta.change(t) != 0 ? true : false

// Check how many bars are in our upper (otf) timeframe
since_new_bar = ta.barssince(is_newbar('D'))
D_total_bars = int(na)
D_total_bars := since_new_bar == 0 ? since_new_bar[1] : D_total_bars[1]

// INDICATORS
HR_short = float(na)
HR_long = float(na)
HR_raw = float(na)
s10 = float(na)
s20 = float(na)

HR_short := barstate.isrealtime ? live_HR_short : hist_HR_short
HR_long := barstate.isrealtime ? live_HR_long : hist_HR_long
HR_raw := barstate.isrealtime ? live_HR_raw : hist_HR_raw

s10 := barstate.isrealtime ? since_new_bar == D_total_bars ? daily_s10 : s10[1] : daily_s10
s20 := barstate.isrealtime ? since_new_bar == D_total_bars ? daily_s20 : s20[1] : daily_s20

capitulation = ta.crossunder(HR_short, HR_long)
miner_capitulation = HR_short < HR_long
recovering = HR_short > HR_short[1] and HR_short > HR_short[2] and HR_short > HR_short[3] and miner_capitulation
recovered = ta.crossover(HR_short, HR_long)

// HASH BOTTOM + PA SIGNAL
buy = false
buy := ta.crossover(s10, s20) and ta.barssince(recovered) < ta.barssince(ta.crossunder(s10, s20)) and ta.barssince(recovered) < ta.barssince(capitulation) or s10 > s20 and ta.crossover(HR_short, HR_long)

buy_plot = buy

// OSCILLATOR
delta = HR_short - HR_long
diff = delta / HR_short * 100

// PLOT - DEFAULT
plot(raw ? HR_raw : na, color=#4caf50, linewidth=1, style=plot.style_line, title='HR Raw')
p1 = plot(type == 'Ribbons' ? HR_long : na, color=#787b86, linewidth=2, style=plot.style_line, title='HR SMA Long')
p2 = plot(type == 'Ribbons' ? HR_short : na, color=HR_short < HR_long ? #ff5252 : #00e676, linewidth=2, style=plot.style_line, title='HR SMA Short')
fill(p1, p2, color=HR_short < HR_long ? #ff52524d : na)

// PLOT - OSCILLATOR
plot(type == 'Oscillator' ? diff : na, style=plot.style_columns, color=diff < 0 ? #ff5252 : #2196f3, title='Oscillator')

// PLOT - SIGNALS
plotshape(signals ? capitulation : na, style=shape.circle, location=location.top, color=#787b8680, size=size.normal, text='Capitulation', textcolor=#363a45, title='Capitulation')
plotshape(signals ? miner_capitulation : na, style=shape.circle, location=location.top, color=#4caf501a, size=size.normal, title='Miner Capitulation')
plotshape(signals ? recovering : na, style=shape.circle, location=location.top, color=#4caf5080, size=size.normal, title='Recovering')
plotshape(signals ? recovered : na, style=shape.circle, location=location.top, color=#00e676, textcolor=#ffffff, size=size.normal, title='Recovered')
plotshape(signals ? buy_plot : na, style=shape.circle, location=location.top, color=#2196f3, text='Buy', textcolor=#2196f3, size=size.normal, title='Buy')

// HALVINGS
halving_1 = timestamp(2012, 11, 28, 0, 0)
halving_2 = timestamp(2016, 7, 9, 0, 0)
halving_3 = timestamp(2020, 4, 30, 0, 0)  // projected! https://www.bitcoinclock.com/
h1_range = time >= halving_1 - 3 * 24 * 60 * 60 * 1000 and time <= halving_1 + 3 * 24 * 60 * 60 * 1000
h2_range = time >= halving_2 - 3 * 24 * 60 * 60 * 1000 and time <= halving_2 + 3 * 24 * 60 * 60 * 1000
h3_range = time >= halving_3 - 3 * 24 * 60 * 60 * 1000 and time <= halving_3 + 3 * 24 * 60 * 60 * 1000
bgcolor(h1_range and plot_halvings ? #ff525233 : na, 90)
bgcolor(h2_range and plot_halvings ? #ff525233 : na, 90)
bgcolor(h3_range and plot_halvings ? #ff525233 : na, 90)

//Strategy
// Conditions
enter_long_condition = ta.crossover(s10, s20) and ta.barssince(recovered) < ta.barssince(ta.crossunder(s10, s20)) and ta.barssince(recovered) < ta.barssince(capitulation) or s10 > s20 and ta.crossover(HR_short, HR_long)
enter_short_condition = capitulation
exit_condition = capitulation

position_signal = 0
position_signal := enter_long_condition ? 1 : enter_short_condition ? -1 : exit_condition ? 0 : position_signal[1]

// Positions
if strategy.opentrades == 0
    contracts = get_round(strategy.equity * trade_leverage / close * (trade_risk / 100), 4)

    if (window() and enter_long_condition)
        strategy.entry("LONG", strategy.long, qty=contracts)
        alert("Long Position Entry Signal")

stop_difference = strategy.position_avg_price / trade_leverage * (trade_stop / 100)

in_long = strategy.position_size > 0

long_stop = in_long ? strategy.position_avg_price - stop_difference : na

strategy.exit('S/L', 'LONG', stop=long_stop, qty=math.abs(get_round(strategy.position_size, 4)))

if (exit_condition or math.abs(ta.change(position_signal)) > 0)
    strategy.close_all()
    alert("Position Exit Signal")
